extern "C" __global__ void
parallelMatricesMultiplication(
    const float *a, 
    const float *b, 
    float *c, 
	int outputRowCount,
	int outputColumnCount,
	int commonDim)
{
	int index = blockDim.x * blockIdx.x + threadIdx.x;
	int outputParamsCount = outputRowCount * outputColumnCount;
	
	if (index < outputParamsCount)
	{
		float sum = 0.0f;
		
		for(int i = 0; i < commonDim; i++)
		{
			int xa = i;
            int ya = index % outputRowCount;

            int xb = index / outputRowCount;
            int yb = i;

            int ia = xa * outputRowCount + ya;
            int ib = xb * commonDim + yb;

            sum += a[ia] * b[ib];
		}
		
		c[index] = sum;
	}
}
